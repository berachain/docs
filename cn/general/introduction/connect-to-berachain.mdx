export const addNetworkMainnet = () => {
  if (typeof window === "undefined" || !window.ethereum) {
    console.warn("[AddNetwork] No wallet detected");
    return;
  }
  const params = {
    chainId: "0x138de",
    chainName: "Berachain",
    nativeCurrency: { name: "BERA Token", symbol: "BERA", decimals: 18 },
    rpcUrls: ["https://rpc.berachain.com/"],
    blockExplorerUrls: ["https://berascan.com/"],
  };
  console.log("[AddNetwork] add mainnet", params);
  window.ethereum
    .request({ method: "wallet_addEthereumChain", params: [params] })
    .then((r) => console.log("[AddNetwork] result", r))
    .catch((e) => console.error("[AddNetwork] error", e));
};

## Berachain 主网 RPC

在支持自定义 RPC 的任何钱包中使用以下参数。一键添加网络，或从下表复制：

  style={{

    alignItems: "center",

    backgroundColor: "#E17726",
    color: "#1a1a1a",
    fontWeight: 600,
    cursor: "pointer",
  }}
>
  添加 Berachain
</button>

| 键         | 值                           |
| ---------- | ---------------------------- |
| 网络名称   | `Berachain`                  |
| RPC URL    | `https://rpc.berachain.com/` |
| Chain ID   | `80094`                      |
| 货币符号   | `BERA`                       |
| 区块浏览器 | `https://berascan.com/`      |

## 支持的钱包

任何支持自定义 RPC 的 [EVM 钱包](https://ethereum.org/en/wallets/find-wallet/)均可用于 Berachain。例如：

- [MetaMask](https://metamask.io)
- [Rabby](https://rabby.io)
- [Coinbase Wallet](https://www.coinbase.com/wallet)
- [Brave Wallet](https://brave.com/wallet/)
- [Frame](https://frame.sh)
- [Zerion](https://zerion.io/download?utm_source=berachain)

## 手动添加 Berachain 网络

若不想使用上方一键按钮，可手动在 MetaMask 中添加网络：

1. **打开网络列表** — 点击 MetaMask 左上角的下拉菜单。

<Frame>
  <img src="/images/general/metamask-add-network-01.png" alt="MetaMask 网络下拉菜单" />
</Frame>

2. **添加网络** — 在弹窗中点击「添加网络」。

<Frame>
  <img src="/images/general/metamask-add-network-02.png" alt="弹窗中的 MetaMask 添加网络按钮" />
</Frame>

3. **手动添加网络** — 在列表底部点击「手动添加网络」。

<Frame>
  <img src="/images/general/metamask-add-network-03.png" alt="MetaMask 手动添加网络选项" />
</Frame>

4. **填写参数** — 使用上方[主网](#berachain-主网-rpc)或[Bepolia 测试网](#berachain-bepolia-测试网-rpc) RPC 表中的参数。

5. **保存** — 点击「保存」。

<Frame>
  <img
    src="/images/general/metamask-add-network-04.png"
    alt="MetaMask 网络配置表单与 Berachain 参数"
  />
</Frame>

此时您应已连接至 Berachain 或 Berachain Bepolia。
