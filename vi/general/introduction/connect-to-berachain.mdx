export const addNetworkMainnet = () => {
  if (typeof window === "undefined" || !window.ethereum) {
    console.warn("[AddNetwork] No wallet detected");
    return;
  }
  const params = {
    chainId: "0x138de",
    chainName: "Berachain",
    nativeCurrency: { name: "BERA Token", symbol: "BERA", decimals: 18 },
    rpcUrls: ["https://rpc.berachain.com/"],
    blockExplorerUrls: ["https://berascan.com/"],
  };
  console.log("[AddNetwork] add mainnet", params);
  window.ethereum.request({ method: "wallet_addEthereumChain", params: [params] }).then(r => console.log("[AddNetwork] result", r)).catch(e => console.error("[AddNetwork] error", e));
};

export const addNetworkBepolia = () => {

## Thêm mạng Berachain thủ công

Nếu không dùng nút một cú nhấp ở trên, thêm mạng trong MetaMask thủ công:

1. **Mở danh sách mạng** — Nhấp vào menu thả xuống ở góc trái trên MetaMask.

<Frame>
  <img src="/images/general/metamask-add-network-01.png" alt="Menu thả xuống mạng MetaMask" />
</Frame>

2. **Thêm mạng** — Nhấp `Add network` trong hộp thoại.

<Frame>
  <img src="/images/general/metamask-add-network-02.png" alt="Nút thêm mạng MetaMask trong hộp thoại" />
</Frame>

3. **Thêm mạng thủ công** — Ở cuối danh sách, nhấp `Add network manually`.

<Frame>
  <img src="/images/general/metamask-add-network-03.png" alt="Tùy chọn thêm mạng thủ công MetaMask" />
</Frame>

4. **Nhập thông tin** — Dùng giá trị từ bảng RPC [Mainnet](#berachain-mainnet-rpc) hoặc [Bepolia Testnet](#berachain-bepolia-testnet-rpc) ở trên.

5. **Lưu** — Nhấp `Save`.

<Frame>
  <img src="/images/general/metamask-add-network-04.png" alt="Biểu mẫu cấu hình mạng MetaMask với thông tin Berachain" />
</Frame>
